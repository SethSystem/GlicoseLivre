    function registrar() {
      const input = document.getElementById('glicose');
      const valor = parseInt(input.value);
      const resultado = document.getElementById('resultado');
      const som = document.getElementById('som-alerta');

      if (isNaN(valor)) {
        resultado.textContent = "Digite um valor vÃ¡lido!";
        resultado.className = "status";
        return;
      }

      let status = '';
      if (valor < 70) {
        status = 'baixo';
        resultado.textContent = "âš ï¸ Glicose Baixa!";
        som.play();
      } else if (valor <= 140) {
        status = 'normal';
        resultado.textContent = "âœ… Glicose Normal.";
      } else {
        status = 'alto';
        resultado.textContent = "âš ï¸ Glicose Alta!";
        som.play();
      }

      resultado.className = `status ${status}`;

      const data = new Date().toLocaleString();
      const historico = JSON.parse(localStorage.getItem('historico')) || [];
      historico.unshift({ valor, data });
      localStorage.setItem('historico', JSON.stringify(historico));
      atualizarHistorico();
      input.value = '';
    }

    function atualizarHistorico() {
      const lista = document.getElementById('lista');
      const historico = JSON.parse(localStorage.getItem('historico')) || [];
      lista.innerHTML = '';
      historico.slice(0, 20).forEach(item => {
        const li = document.createElement('li');
        li.textContent = `${item.data} - ${item.valor} mg/dL`;
        lista.appendChild(li);
      });
    }

    atualizarHistorico();
  </script>
